Integration Quickstart

docker pull hrco/xbridge-mcp:latest

docker run -e XAI_API_KEY=your_key_here hrco/xbridge-mcp:latest
from mcp import Client

client = Client("xbridge")
resp = client.call("grok-chat", {"message": "Summarize this repo"})
print(resp)
const res = await mcp.call("grok-web-search", {
  query: "latest xAI Grok updates"
});
console.log(res);
